

Table Company_master {
  Company_ID int [pk, increment]
  Company_Name varchar(255) [not null]
  Legal_Name varchar(255)
  GSTIN varchar(15) [unique]
  PAN varchar(10) [unique]
  CIN varchar(20)
  Business_Type varchar(100)
  Address_Line_1 varchar(500)
  Address_Line_2 varchar(500)
  City varchar(100)
  State varchar(100)
  State_Code varchar(10)
  Pincode varchar(10)
  Country varchar(100)
  Phone varchar(20)
  Email varchar(255)
  Financial_Year_From date
  Financial_Year_To date
  Currency varchar(3) [default: 'INR']
  Books_Beginning_From date
  GST_Applicable boolean [default: true]
  Composition_Scheme boolean [default: false]
  Created_Date datetime [default: `now()`]
  Status varchar(20) [default: 'Active']
}

Table Ledger_Master {
  Ledger_ID int [pk, increment]
  Company_ID int [ref: > Company_master.Company_ID]
  Ledger_Name varchar(255) [not null]
  Alias_Name varchar(255)
  Parent_Group varchar(255)
  Group_Type varchar(100)
  Opening_Balance decimal(15,2) [default: 0]
  Opening_Balance_Type varchar(10) [default: 'Dr']
  GST_Applicable boolean [default: false]
  GST_Tax_Type varchar(50)
  GST_Rate decimal(5,2) [default: 0]
  HSN_SAC varchar(20)
  Is_Bill_Wise boolean [default: false]
  Credit_Limit decimal(15,2) [default: 0]
  Maintain_Balances_Bill_Wise boolean [default: false]
  Status varchar(20) [default: 'Active']
}

Table Party_Master {
  Party_ID int [pk, increment]
  Company_ID int [ref: > Company_master.Company_ID]
  Ledger_ID int [ref: > Ledger_Master.Ledger_ID]
  Party_Type varchar(50) [not null]
  Party_Name varchar(255) [not null]
  GSTIN varchar(15)
  PAN varchar(10)
  Contact_Person varchar(255)
  Phone varchar(20)
  Email varchar(255)
  Billing_Address text
  Shipping_Address text
  State varchar(100)
  State_Code varchar(10)
  Credit_Period int [default: 30]
  Credit_Limit decimal(15,2) [default: 0]
  Opening_Balance decimal(15,2) [default: 0]
  Balance_Type varchar(10) [default: 'Dr']
  TDS_Applicable boolean [default: false]
  Status varchar(20) [default: 'Active']
}

Table Item_Master {
  Item_ID int [pk, increment]
  Company_ID int [ref: > Company_master.Company_ID]
  Item_Name varchar(255) [not null]
  Alias_Name varchar(255)
  Category varchar(100)
  Unit varchar(20)
  HSN_Code varchar(20)
  GST_Rate decimal(5,2) [default: 0]
  Purchase_Rate decimal(12,2) [default: 0]
  Sales_Rate decimal(12,2) [default: 0]
  MRP decimal(12,2) [default: 0]
  Opening_Stock decimal(12,3) [default: 0]
  Opening_Stock_Value decimal(15,2) [default: 0]
  Reorder_Level decimal(12,3) [default: 0]
  Batch_Applicable boolean [default: false]
  Expiry_Date date
  Status varchar(20) [default: 'Active']
}

Table Employee_Master {
  Employee_ID int [pk, increment]
  Company_ID int [ref: > Company_master.Company_ID]
  Employee_Name varchar(255) [not null]
  Department varchar(100)
  Designation varchar(100)
  Date_of_Joining date [not null]
  Mobile varchar(15)
  Email varchar(255)
  Salary decimal(12,2) [default: 0]
  Attendance_Applicable boolean [default: true]
  PF_Applicable boolean [default: false]
  ESI_Applicable boolean [default: false]
  Status varchar(20) [default: 'Active']
}

Table Sales_Voucher {
  Sales_Voucher_No varchar(50) [pk]
  Company_ID int [ref: > Company_master.Company_ID]
  Voucher_Date date [not null]
  Party_ID int [ref: > Party_Master.Party_ID]
  Invoice_No varchar(50)
  Invoice_Date date
  GST_Type varchar(50)
  Total_Tax decimal(15,2)
  Round_Off decimal(10,2) [default: 0]
  Total_Invoice_Value decimal(15,2) [not null]
  Payment_Status varchar(50) [default: 'Pending']
  Created_By int [ref: > Employee_Master.Employee_ID]
}

Table Purchase_Voucher {
  Purchase_Voucher_No varchar(50) [pk]
  Company_ID int [ref: > Company_master.Company_ID]
  Voucher_Date date [not null]
  Vendor_ID int [ref: > Party_Master.Party_ID]
  Invoice_No varchar(50)
  Invoice_Date date
  Total_Tax decimal(15,2)
  Total_Amount decimal(15,2) [not null]
  Reverse_Charge boolean [default: false]
  Input_Tax_Credit decimal(15,2) [default: 0]
  Payment_Status varchar(50) [default: 'Pending']
}

Table Voucher_Line_Items {
  Line_ID int [pk, increment]
  Company_ID int [ref: > Company_master.Company_ID]
  Voucher_Type varchar(20) [not null] // 'Sales' or 'Purchase'
  Sales_Voucher_No varchar(50) 
  Purchase_Voucher_No varchar(50)
  Item_ID int [ref: > Item_Master.Item_ID]
  Quantity decimal(12,3) [not null]
  Rate decimal(12,2) [not null]
  Discount decimal(6,2) [default: 0]
  Taxable_Value decimal(15,2) [not null]
  CGST_Rate decimal(5,2)
  CGST_Amount decimal(12,2)
  SGST_Rate decimal(5,2)
  SGST_Amount decimal(12,2)
  IGST_Rate decimal(5,2)
  IGST_Amount decimal(12,2)
  Line_Total decimal(15,2)
}

Table Payment_Receipt {
  Transaction_ID int [pk, increment]
  Company_ID int [ref: > Company_master.Company_ID]
  Transaction_Type varchar(50) [not null]
  Voucher_No varchar(50)
  Voucher_Date date
  Party_ID int [ref: > Party_Master.Party_ID]
  Payment_Mode varchar(50) [not null]
  Bank_Name varchar(255)
  Reference_No varchar(100)
  Amount decimal(15,2) [not null]
  Narration text
  Created_By int [ref: > Employee_Master.Employee_ID]
}

Table Expense_Voucher {
  Expense_ID int [pk, increment]

Table Employee_Attendance {
  Attendance_ID int [pk, increment]
  Employee_ID int [ref: > Employee_Master.Employee_ID]
  Date date [not null]
  Punch_In_Time time
  Punch_Out_Time time
  Working_Hours decimal(5,2)
  Overtime_Hours decimal(5,2) [default: 0]

Table GST_Report_Data {
  GSTIN varchar(15) [pk]
  Period varchar(10) [pk] // YYYY-MM
  Return_Type varchar(50) [pk]
  Company_ID int [ref: > Company_master.Company_ID]
  Taxable_Value decimal(15,2)
  CGST decimal(15,2)
  SGST decimal(15,2)
  IGST decimal(15,2)
  Total_Tax_Liability decimal(15,2)
  ITC_Claimed decimal(15,2)
  Net_Tax_Payable decimal(15,2)
}

Table Documents {
  Document_ID int [pk, increment]
  Company_ID int [ref: > Company_master.Company_ID]
  Document_Type varchar(100)
  Linked_Table varchar(50) // Manual polymorphic: 'Sales_Voucher', etc.
  Linked_ID varchar(50) [not null]
  Uploaded_By int [ref: > Employee_Master.Employee_ID]
  Upload_Date datetime [default: `now()`]
  File_Name varchar(255) [not null]
  File_URL varchar(500)
}

Table Digital_Sign {
  Sign_ID int [pk, increment]
  Document_ID int [ref: > Documents.Document_ID]
  Signed_By varchar(255) [not null]
  Sign_Date datetime [default: `now()`]
  Sign_Status varchar(50) [default: 'Pending']
  IP_Address varchar(45)
}
